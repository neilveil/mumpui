MumpUI / Setup

Step I - Install

npm i mumpui

Step II - Initialize MumpUI

MumpUI need to be initialized only once at the top.
import MumpUI from 'mumpui'

function App() {
  return (<div>
    {/* Initialize MumpUI */}
    <MumpUI />

    ..

    <BrowserRouter basename="/app">
      <Routes>
        <Route path="/" />
        ..
      </Routes>
    </BrowserRouter>
  </div>)
}

createRoot(document.getElementById('root') as HTMLElement).render(<App />)
